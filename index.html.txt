        <button class="action-btn btn-submit" id="submit-btn" onclick="sendToLine()">送出</button>
    </div>

    <script>
        // 1. 初始化
        const grid = document.getElementById('grid');
        const countText = document.getElementById('count-text');
        const submitBtn = document.getElementById('submit-btn');
        let selectedNumbers = new Set(); // 使用 Set 來儲存選中的號碼，避免重複

        // 2. 自動產生 1-39 的按鈕
        for (let i = 1; i <= 39; i++) {
            let btn = document.createElement('div');
            // 補零格式，例如 5 變成 '05'
            let numStr = i.toString().padStart(2, '0');
            btn.className = 'num-btn';
            btn.textContent = numStr;
            btn.dataset.value = numStr;
            
            // 點擊事件
            btn.onclick = () => toggleNumber(btn, numStr);
            grid.appendChild(btn);
        }

        // 3. 切換號碼選取狀態
        function toggleNumber(btn, num) {
            if (selectedNumbers.has(num)) {
                // 如果已經選過，則取消
                selectedNumbers.delete(num);
                btn.classList.remove('selected');
            } else {
                // 限制最多選 5 個
                if (selectedNumbers.size >= 5) {
                    alert('阿公/阿嬤，只能選 5 個號碼喔！\n請先取消一個再選。');
                    return;
                }
                // 選取
                selectedNumbers.add(num);
                btn.classList.add('selected');
            }
            updateUI();
        }

        // 4. 更新介面顯示 (文字與按鈕狀態)
        function updateUI() {
            const count = selectedNumbers.size;
            countText.textContent = `請選 5 個號碼 (${count}/5)`;
            
            // 只有選滿 5 個時，送出按鈕才亮起
            if (count === 5) {
                submitBtn.classList.add('active');
                submitBtn.textContent = "確認送出";
                countText.style.color = "green";
                countText.textContent = "選好了！請按送出";
            } else {
                submitBtn.classList.remove('active');
                submitBtn.textContent = "還沒選完";
                countText.style.color = "#666";
            }
        }

        // 5. 重置所有選項
        function resetSelection() {
            selectedNumbers.clear();
            document.querySelectorAll('.num-btn').forEach(btn => {
                btn.classList.remove('selected');
            });
            updateUI();
        }

        // 6. 初始化 LIFF 並送出訊息
        // 記得將 YOUR_LIFF_ID 換成你申請的 ID
        liff.init({ liffId: "YOUR_LIFF_ID" }).then(() => {
            console.log("LIFF Initialized");
        }).catch(err => {
            console.error(err);
        });

        function sendToLine() {
            if (selectedNumbers.size !== 5) return;

            // 將 Set 轉為陣列並排序 (小到大)
            const sortedNums = Array.from(selectedNumbers).sort((a, b) => a - b);
            const msgText = `我要下注：${sortedNums.join(', ')}`;

            if (liff.isInClient()) {
                liff.sendMessages([{
                    type: 'text',
                    text: msgText
                }]).then(() => {
                    liff.closeWindow(); // 發送成功後關閉視窗
                }).catch((err) => {
                    alert('發送失敗，請重試');
                });
            } else {
                alert('這個功能只能在 LINE 裡面使用喔！\n您選的是：' + msgText);
            }
        }
    </script>
